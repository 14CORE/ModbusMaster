68 static uint16_t crc16_update(uint16_t crc, uint8_t a)
69 {
70  int i;
71 
72  crc ^= a;
73  for (i = 0; i < 8; ++i)
74  {
75  if (crc & 1)
76  crc = (crc >> 1) ^ 0xA001;
77  else
78  crc = (crc >> 1);
79  }
80 
81  return crc;
82 }
